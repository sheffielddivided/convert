async function searchCompanies(){ const q = document .getElementById("query") .value .trim(); if(!q) return; status.textContent = "Searching..."; results.innerHTML = ""; try{ const url = `https://data.gov.sg/api/action/package_show?id=d_3f960c10fed6145404ca7b821f263b87`; const meta = await fetch(url) .then(r=>r.json()); const resource = meta .result .resources[0] .id; const searchUrl = `https://data.gov.sg/api/action/datastore_search` + `?resource_id=${resource}` + `&limit=10` + `&fields=entity_name,uen,uen_status_desc,entity_type_desc` + `&filters=` + encodeURIComponent( JSON.stringify({ entity_name:q }) ); const res = await fetch( searchUrl ); const json = await res.json(); const rows = json .result .records || []; render( rows ); status.textContent = `${rows.length} results`; } catch(e){ console.error(e); status.textContent = "Search failed"; } } function render(rows){ results.innerHTML=""; rows.forEach(row=>{ const div = document.createElement( "div" ); div.className = "card"; div.innerHTML = ` ${row.entity_name}
UEN: ${row.uen}
Status: ${row.uen_status_desc}
Type: ${row.entity_type_desc}
`; results.appendChild( div); }); }
